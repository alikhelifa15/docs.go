# Interfaces en Go

## Concepts de base

### Qu'est-ce qu'une interface ?
Une interface d√©finit un ensemble de m√©thodes. Tout type qui impl√©mente ces m√©thodes satisfait automatiquement l'interface.


## Patterns d'interfaces

### Strategy Pattern
```go
package main

import "fmt"

// Interface pour les strat√©gies de tri
type SortStrategy interface {
    Sort([]int) []int
}

// Tri √† bulles
type BubbleSort struct{}

func (bs BubbleSort) Sort(data []int) []int {
    arr := make([]int, len(data))
    copy(arr, data)
    
    n := len(arr)
    for i := 0; i < n-1; i++ {
        for j := 0; j < n-i-1; j++ {
            if arr[j] > arr[j+1] {
                arr[j], arr[j+1] = arr[j+1], arr[j]
            }
        }
    }
    fmt.Println("Utilis√©: Bubble Sort")
    return arr
}

// Tri par s√©lection
type SelectionSort struct{}

func (ss SelectionSort) Sort(data []int) []int {
    arr := make([]int, len(data))
    copy(arr, data)
    
    n := len(arr)
    for i := 0; i < n-1; i++ {
        minIdx := i
        for j := i + 1; j < n; j++ {
            if arr[j] < arr[minIdx] {
                minIdx = j
            }
        }
        arr[i], arr[minIdx] = arr[minIdx], arr[i]
    }
    fmt.Println("Utilis√©: Selection Sort")
    return arr
}

// Context qui utilise la strat√©gie
type Sorter struct {
    strategy SortStrategy
}

func (s *Sorter) SetStrategy(strategy SortStrategy) {
    s.strategy = strategy
}

func (s *Sorter) Sort(data []int) []int {
    if s.strategy == nil {
        fmt.Println("Aucune strat√©gie d√©finie")
        return data
    }
    return s.strategy.Sort(data)
}

func main() {
    data := []int{64, 34, 25, 12, 22, 11, 90}
    sorter := &Sorter{}
    
    fmt.Printf("Donn√©es originales: %v\n", data)
    
    // Utiliser bubble sort
    sorter.SetStrategy(BubbleSort{})
    result1 := sorter.Sort(data)
    fmt.Printf("Bubble sort: %v\n", result1)
    
    // Changer pour selection sort
    sorter.SetStrategy(SelectionSort{})
    result2 := sorter.Sort(data)
    fmt.Printf("Selection sort: %v\n", result2)
}
```

### Observer Pattern

### Interface avec contraintes g√©n√©riques (Go 1.18+)

```go
package main

import "fmt"

// Interface pour types ordonnables
type Ordered interface {
    ~int | ~int8 | ~int16 | ~int32 | ~int64 |
    ~uint | ~uint8 | ~uint16 | ~uint32 | ~uint64 |
    ~float32 | ~float64 | ~string

// Fonction g√©n√©rique qui utilise l'interface
func Min[T Ordered](a, b T) T {
    if a < b {
        return a
    }
    return b
}

    if a > b {
        return a
    }
    return b

// Interface plus complexe
type Comparable[T any] interface {
    CompareTo(other T) int  // -1: plus petit, 0: √©gal, 1: plus grand
}

type Version struct {
    Major, Minor, Patch int
}

func (v Version) CompareTo(other Version) int {
    if v.Major != other.Major {
        if v.Major < other.Major {
            return -1
        }
        return 1
    }
    if v.Minor != other.Minor {
        if v.Minor < other.Minor {
            return -1
        }
        return 1
    }
    if v.Patch != other.Patch {
        if v.Patch < other.Patch {
            return -1
        }
        return 1
    }
    return 0
}

func (v Version) String() string {
    return fmt.Sprintf("v%d.%d.%d", v.Major, v.Minor, v.Patch)
}

func FindMax[T Comparable[T]](items []T) T {
    if len(items) == 0 {
        var zero T
        return zero
    }
    
    max := items[0]
    for _, item := range items[1:] {
        if item.CompareTo(max) > 0 {
            max = item
        }
    }
    return max
}

    // Utilisation avec type personnalis√©
    versions := []Version{
        {1, 2, 3},
        {2, 0, 1},
        {1, 5, 0},
        {2, 1, 0},
    }
```

### Interface pour middleware HTTP
```go
package main

import (
    "fmt"
    "log"
    "net/http"
    "time"
)

// Interface pour middleware
type Middleware interface {
    Handle(http.HandlerFunc) http.HandlerFunc
}

// Middleware de logging
type LoggingMiddleware struct{}

func (lm LoggingMiddleware) Handle(next http.HandlerFunc) http.HandlerFunc {
    return func(w http.ResponseWriter, r *http.Request) {
        start := time.Now()
        log.Printf("D√©but %s %s", r.Method, r.URL.Path)
        
        next(w, r)
        
        duration := time.Since(start)
        log.Printf("Fin %s %s (dur√©e: %v)", r.Method, r.URL.Path, duration)
    }
}

// Middleware d'authentification
type AuthMiddleware struct {
    validTokens map[string]bool
}

func NewAuthMiddleware() *AuthMiddleware {
    return &AuthMiddleware{
        validTokens: map[string]bool{
            "token123": true,
            "secret":   true,
        },
    }
}

func (am *AuthMiddleware) Handle(next http.HandlerFunc) http.HandlerFunc {
    return func(w http.ResponseWriter, r *http.Request) {
        token := r.Header.Get("Authorization")
        if token == "" {
            http.Error(w, "Token manquant", http.StatusUnauthorized)
            return
        }
        
        if !am.validTokens[token] {
            http.Error(w, "Token invalide", http.StatusUnauthorized)
            return
        }
        
        next(w, r)
    }
}

// Middleware de rate limiting
type RateLimitMiddleware struct {
    requests map[string]int
    limit    int
}

func NewRateLimitMiddleware(limit int) *RateLimitMiddleware {
    return &RateLimitMiddleware{
        requests: make(map[string]int),
        limit:    limit,
    }
}

func (rlm *RateLimitMiddleware) Handle(next http.HandlerFunc) http.HandlerFunc {
    return func(w http.ResponseWriter, r *http.Request) {
        ip := r.RemoteAddr
        rlm.requests[ip]++
        
        if rlm.requests[ip] > rlm.limit {
            http.Error(w, "Rate limit d√©pass√©", http.StatusTooManyRequests)
            return
        }
        
        next(w, r)
    }
}

// Gestionnaire de middleware
type MiddlewareStack struct {
    middlewares []Middleware
}

func NewMiddlewareStack() *MiddlewareStack {
    return &MiddlewareStack{}
}

func (ms *MiddlewareStack) Use(middleware Middleware) {
    ms.middlewares = append(ms.middlewares, middleware)
}

func (ms *MiddlewareStack) Apply(handler http.HandlerFunc) http.HandlerFunc {
    result := handler
    
    // Appliquer les middlewares en ordre inverse
    for i := len(ms.middlewares) - 1; i >= 0; i-- {
        result = ms.middlewares[i].Handle(result)
    }
    
    return result
}

// Handler d'exemple
func helloHandler(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintf(w, "Bonjour! Il est %s\n", time.Now().Format("15:04:05"))
}

func main() {
    stack := NewMiddlewareStack()
    
    // Ajouter les middlewares
    stack.Use(LoggingMiddleware{})
    stack.Use(NewAuthMiddleware())
    stack.Use(NewRateLimitMiddleware(5))
    
    // Appliquer la stack au handler
    protectedHandler := stack.Apply(helloHandler)
    
    http.HandleFunc("/hello", protectedHandler)
    
    fmt.Println("Serveur d√©marr√© sur :8080")
    fmt.Println("Testez avec: curl -H 'Authorization: token123' http://localhost:8080/hello")
    
    log.Fatal(http.ListenAndServe(":8080", nil))
}
```
