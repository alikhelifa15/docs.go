# Gestion d'erreurs en Go

Go adopte une approche explicite pour la gestion d'erreurs. Pas d'exceptions, mais des valeurs d'erreur qui doivent √™tre v√©rifi√©es explicitement.

## Concepts de base

### L'interface error
```go
// L'interface error est pr√©d√©finie dans Go

    Error() string
}
```

### Cr√©ation d'erreurs simples
```go
package main

    "errors"
    "fmt"
)

    if b == 0 {
        // Cr√©er une erreur simple
        return 0, errors.New("division par z√©ro")
    }
    return a / b, nil

    if x < 0 {
        // Erreur format√©e
        return 0, fmt.Errorf("impossible de calculer la racine de %g", x)
    }
    return math.Sqrt(x), nil

    // Pattern classique de gestion d'erreur
    resultat, err := diviser(10, 0)
    if err != nil {
        fmt.Printf("Erreur: %v\n", err)

    }
    fmt.Printf("R√©sultat: %g\n", resultat)
    
    // Avec une fonction qui r√©ussit

        fmt.Printf("Erreur: %v\n", err)
```

```go
            Field:     "email",
            Value:     email,

        Op:        "GET",
        URL:       url,
        Timestamp: time.Now(),

    }

        fmt.Printf("‚Üí Heure: %s\n", e.Timestamp.Format("15:04:05"))

        fmt.Printf("‚Üí URL: %s\n", e.URL)
        fmt.Printf("‚Üí Tentatives: %d\n", e.Retries)
    }
}
```

## Patterns de gestion d'erreurs

### Retry avec backoff
```go
package main

import (
    "errors"
    "fmt"
    "math/rand"
    "time"
)

type RetryConfig struct {
    MaxAttempts int
    BaseDelay   time.Duration
    MaxDelay    time.Duration
    Multiplier  float64
}

func DefaultRetryConfig() RetryConfig {
    return RetryConfig{
        MaxAttempts: 3,
        BaseDelay:   100 * time.Millisecond,
        MaxDelay:    5 * time.Second,
        Multiplier:  2.0,
    }
}

// Interface pour erreurs temporaires
type TemporaryError interface {
    error
    IsTemporary() bool
}

type ServiceError struct {
    message   string
    temporary bool
    code      int
}

func (e ServiceError) Error() string {
    return e.message
}

func (e ServiceError) IsTemporary() bool {
    return e.temporary
}

// Fonction qui peut √©chouer temporairement
func unreliableService() error {
    // 70% de chance d'√©chouer temporairement
    if rand.Float32() < 0.7 {
        return ServiceError{
            message:   "service temporairement indisponible",
            temporary: true,
            code:      503,
        }
    }
    
    // 10% de chance d'√©chouer d√©finitivement
    if rand.Float32() < 0.1 {
        return ServiceError{
            message:   "erreur de configuration",
            temporary: false,
            code:      500,
        }
    }
    
    return nil // Succ√®s
}

func retryWithBackoff(operation func() error, config RetryConfig) error {
    var lastErr error
    delay := config.BaseDelay
    
    for attempt := 1; attempt <= config.MaxAttempts; attempt++ {
        fmt.Printf("Tentative %d/%d...\n", attempt, config.MaxAttempts)
        
        err := operation()
        if err == nil {
            fmt.Printf("‚úÖ Succ√®s √† la tentative %d\n", attempt)
            return nil
        }
        
        lastErr = err
        
        // V√©rifier si l'erreur est temporaire
        if tempErr, ok := err.(TemporaryError); ok && !tempErr.IsTemporary() {
            fmt.Printf("‚ùå Erreur permanente: %v\n", err)
            return err
        }
        
        if attempt < config.MaxAttempts {
            fmt.Printf("‚è≥ √âchec, retry dans %v (erreur: %v)\n", delay, err)
            time.Sleep(delay)
            
            // Exponential backoff avec jitter
            delay = time.Duration(float64(delay) * config.Multiplier)
            if delay > config.MaxDelay {
                delay = config.MaxDelay
            }
            
            // Ajouter du jitter (¬±25%)
            jitter := time.Duration(rand.Float64() * float64(delay) * 0.5)
            delay = delay + jitter - jitter/2
        }
    }
    
    return fmt.Errorf("√©chec apr√®s %d tentatives, derni√®re erreur: %w", config.MaxAttempts, lastErr)
}

func main() {
    rand.Seed(time.Now().UnixNano())
    
    config := DefaultRetryConfig()
    
    fmt.Println("=== Test de retry avec backoff ===")
    err := retryWithBackoff(unreliableService, config)
    if err != nil {
        fmt.Printf("üí• √âchec final: %v\n", err)
    }
}
```

### Circuit Breaker pattern
```go
package main

import (
    "errors"
    "fmt"
    "sync"
    "time"
)

type CircuitState int

const (
    Closed CircuitState = iota
    Open
    HalfOpen
)

func (s CircuitState) String() string {
    switch s {
    case Closed:
        return "CLOSED"
    case Open:
        return "OPEN"
    case HalfOpen:
        return "HALF_OPEN"
    default:
        return "UNKNOWN"
    }
}

type CircuitBreakerConfig struct {
    FailureThreshold int
    RecoveryTimeout  time.Duration
    SuccessThreshold int
}

type CircuitBreaker struct {
    config           CircuitBreakerConfig
    state            CircuitState
    failures         int
    successes        int
    lastFailureTime  time.Time
    mu               sync.RWMutex
}

var (
    ErrCircuitOpen = errors.New("circuit breaker ouvert")
)

func NewCircuitBreaker(config CircuitBreakerConfig) *CircuitBreaker {
    return &CircuitBreaker{
        config: config,
        state:  Closed,
    }
}

func (cb *CircuitBreaker) Execute(operation func() error) error {
    // V√©rifier l'√©tat avant ex√©cution
    if !cb.canExecute() {
        return ErrCircuitOpen
    }
    
    // Ex√©cuter l'op√©ration
    err := operation()
    
    // Mettre √† jour l'√©tat en fonction du r√©sultat
    cb.recordResult(err)
    
    return err
}

func (cb *CircuitBreaker) canExecute() bool {
    cb.mu.RLock()
    defer cb.mu.RUnlock()
    
    switch cb.state {
    case Closed:
        return true
    case Open:
        // V√©rifier si on peut passer en half-open
        if time.Since(cb.lastFailureTime) > cb.config.RecoveryTimeout {
            return true
        }
        return false
    case HalfOpen:
        return true
    default:
        return false
    }
}

func (cb *CircuitBreaker) recordResult(err error) {
    cb.mu.Lock()
    defer cb.mu.Unlock()
    
    if err != nil {
        cb.recordFailure()
    } else {
        cb.recordSuccess()
    }
}

func (cb *CircuitBreaker) recordFailure() {
    cb.failures++
    cb.successes = 0
    cb.lastFailureTime = time.Now()
    
    switch cb.state {
    case Closed:
        if cb.failures >= cb.config.FailureThreshold {
            cb.state = Open
            fmt.Printf("üî¥ Circuit breaker OUVERT (%d √©checs)\n", cb.failures)
        }
    case HalfOpen:
        cb.state = Open
        fmt.Printf("üî¥ Circuit breaker OUVERT (√©chec en half-open)\n")
    }
}

func (cb *CircuitBreaker) recordSuccess() {
    cb.failures = 0
    cb.successes++
    
    switch cb.state {
    case Open:
        cb.state = HalfOpen
        cb.successes = 1
        fmt.Printf("üü° Circuit breaker HALF-OPEN\n")
    case HalfOpen:
        if cb.successes >= cb.config.SuccessThreshold {
            cb.state = Closed
            fmt.Printf("üü¢ Circuit breaker FERM√â\n")
        }
    }
}

func (cb *CircuitBreaker) GetState() CircuitState {
    cb.mu.RLock()
    defer cb.mu.RUnlock()
    return cb.state
}

func (cb *CircuitBreaker) GetStats() (CircuitState, int, int) {
    cb.mu.RLock()
    defer cb.mu.RUnlock()
    return cb.state, cb.failures, cb.successes
}

// Service simul√© qui peut √©chouer
func flakyService(failureRate float32) func() error {
    return func() error {
        if rand.Float32() < failureRate {
            return errors.New("service indisponible")
        }
        return nil
    }
}

func main() {
    rand.Seed(time.Now().UnixNano())
    
    config := CircuitBreakerConfig{
        FailureThreshold: 3,
        RecoveryTimeout:  2 * time.Second,
        SuccessThreshold: 2,
    }
    
    cb := NewCircuitBreaker(config)
    
    // Service qui √©choue 80% du temps initialement
    service := flakyService(0.8)
    
    fmt.Println("=== Test Circuit Breaker ===")
    
    // Phase 1: Provoquer l'ouverture du circuit
    fmt.Println("\nPhase 1: Provocation d'√©checs...")
    for i := 0; i < 10; i++ {
        err := cb.Execute(service)
        state, failures, successes := cb.GetStats()
        
        if err != nil {
            if errors.Is(err, ErrCircuitOpen) {
                fmt.Printf("%d. ‚ùå Circuit ouvert, requ√™te bloqu√©e\n", i+1)
            } else {
                fmt.Printf("%d. ‚ùå Service failed: %v\n", i+1, err)
            }
        } else {
            fmt.Printf("%d. ‚úÖ Service r√©ussi\n", i+1)
        }
        
        fmt.Printf("   √âtat: %s, √âchecs: %d, Succ√®s: %d\n", state, failures, successes)
        time.Sleep(200 * time.Millisecond)
    }
    
    // Phase 2: Attendre et am√©liorer le service
    fmt.Println("\nPhase 2: Attente de recovery...")
    time.Sleep(3 * time.Second)
    
    // Am√©liorer le service (seulement 20% d'√©chec)
    service = flakyService(0.2)
    
    fmt.Println("\nPhase 3: Test de r√©cup√©ration...")
    for i := 0; i < 10; i++ {
        err := cb.Execute(service)
        state, failures, successes := cb.GetStats()
        
        if err != nil {
            if errors.Is(err, ErrCircuitOpen) {
                fmt.Printf("%d. ‚ùå Circuit ouvert, requ√™te bloqu√©e\n", i+1)
            } else {
                fmt.Printf("%d. ‚ùå Service failed: %v\n", i+1, err)
            }
        } else {
            fmt.Printf("%d. ‚úÖ Service r√©ussi\n", i+1)
        }
        
        fmt.Printf("   √âtat: %s, √âchecs: %d, Succ√®s: %d\n", state, failures, successes)
        time.Sleep(500 * time.Millisecond)
    }
}
```

### Erreurs avec stack trace

```go
package main

import (
    "fmt"
    "runtime"
    "strings"
)

type ErrorWithStack struct {
    message string
    stack   []uintptr
}

func NewErrorWithStack(message string) *ErrorWithStack {
    const depth = 32
    var pcs [depth]uintptr
    n := runtime.Callers(2, pcs[:])
    
    return &ErrorWithStack{
        message: message,
        stack:   pcs[0:n],
    }
}

func (e *ErrorWithStack) Error() string {
    return e.message
}

func (e *ErrorWithStack) StackTrace() string {
    var lines []string
    frames := runtime.CallersFrames(e.stack)
    
    for {
        frame, more := frames.Next()
        if !more {
            break
        }
        
        line := fmt.Sprintf("%s\n\t%s:%d", frame.Function, frame.File, frame.Line)
        lines = append(lines, line)
    }
    
    return strings.Join(lines, "\n")
}

    return NewErrorWithStack("erreur au niveau 3")
}

        return fmt.Errorf("erreur au niveau 2: %w", err)
    }
    return nil
}

    }
    return nil
}

        fmt.Printf("Erreur: %v\n", err)
        
        // Extraire le stack trace
        var stackErr *ErrorWithStack
```
